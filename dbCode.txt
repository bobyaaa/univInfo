CREATE DATABASE univinfo;
USE univinfo;
CREATE TABLE universities(
  univ_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
  univ_name VARCHAR(40) NOT NULL,
  overview VARCHAR(2000) NULL,
  famous VARCHAR(2000) NULL
);
CREATE TABLE programs(
  prog_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
  prog_name VARCHAR(40) NOT NULL,
  univ_id INT UNSIGNED NOT NULL,
  info VARCHAR(2000) NULL
);
